1 // extern
  2 var Texture;
  3 
  4 function TextureManager() {
  5     this.cache = {};
  6 }
  7 
  8 
  9 TextureManager.prototype = {
 10     constructor: TextureManager,
 11     load: function ( url ) {
 12         if ( this.cache[ url ] )  {
 13             return this.cache[ url ];
 14         }
 15 
 16         var texture = new Texture( url.split( '/' ).pop() );
 17 
 18         var img = new Image();
 19         img.onload = function() {
 20             texture.setImage( img );
 21         };
 22         img.src = url;
 23 
 24         this.cache[ url ] = texture;
 25 
 26         return texture;
 27     }
 28 };
