  2 var Camera, RenderManager, Scene, Importer, Exporter, InputHandler;
  3 
  4 function Application() {
  5     this.renderManager = new RenderManager();
  6     this.scene = new Scene();
  7     this.camera = new Camera().setPosition( [ 0, 0, 10 ] );
  8 
  9     this.importer = new Importer( 'resources' );
 10     this.exporter = new Exporter( 'resources' );
 11 
 12     this.input = new InputHandler();
 13 
 14     this.scene.appendChild( this.camera );
 15 
 16     var canvas = this.renderManager.renderer.canvas;
 17     this.setupCanvas( canvas );
 18 
 19     var self = this;
 20     this._nextFrame = null;
 21     this.capFPS( 60 );
 22 
 23     var t = Date.now();
 24     function renderLoop() {
 25         var now = Date.now();
 26         self.onBeforeRender( t - now );
 27         t = now;
 28         self.renderManager.renderScene( self.scene, self.camera );
 29         // console.log( self._nextFrame );
 30         self._nextFrame( renderLoop );
 31     }
 32 
 33     // it is necessary to call this asynchronously because the inheriting
 34     // developer may override the constructor and will call the parent constructor
 35     // initially; the rest of the inheriting constructor will initilize some objects
 36     // which may be required by the renderLoop; therefore run it after the inheriting
 37     // constructor has finished initializations
 38     setTimeout( renderLoop, 1 );
 39 }
 40 
 41 Application.prototype = {
 42     constructor: Application,
 43     setupCanvas: function( canvas ) {
 44         var self = this;
 45         window.addEventListener( 'resize', function() {
 46             self.resize( window.innerWidth, window.innerHeight );
 47         }, false );
 48         document.body.appendChild( canvas );
 49         this.resize( window.innerWidth, window.innerHeight );
 50         return this;
 51     },
 52     resize: function( width, height ) {
 53         this.renderManager.resize( width, height );
 54         var camera = this.camera;
 55         camera.width = width;
 56         camera.height = height;
 57         camera.setPerspective();
 58         return this;
 59     },
 60     onBeforeRender: function ( elapsed ) {
 61         // overwrite me
 62     },
 63     capFPS: function( fps ) {
 64         if ( fps >= 60 ) {
 65             this._nextFrame = window.requestAnimationFrame.bind( window );
 66         }
 67         else {
 68             this._nextFrame = function( renderLoop ) {
 69                 setTimeout( renderLoop, 1000 / fps );
 70             };
 71         }
 72     }
 73 };
