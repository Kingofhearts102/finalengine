1 // extern
  2 var Matrix4, Renderer, Drawable, Framebuffer, Mesh, Buffer, VertexAttribute, Light, Texture;
  3 
  4 function RenderManager() {
  5     this.renderer = new Renderer();
  6     this.forcedMaterial = null;
  7 
  8     this.postProcess = false;
  9 
 10     this.framebuffer = new Framebuffer( this.renderer.width, this.renderer.height );
 11 
 12     if ( this.renderer.getParameter( Renderer.FLOAT_TEXTURE ) ) {
 13         this.framebuffer.colorTexture.setDataType( Texture.FLOAT );
 14     }
 15 
 16     this.postProcessEffects = [];
 17 
 18     this.quad = new Mesh();
 19     this.quad.setVertexAttribute( new VertexAttribute( 'UVCoord' ).setBuffer( new Buffer().setData( [ -1, -1, 1, 1, -1, 1, 1, -1 ] ) ).setSize( 2 ) );
 20     this.quad.setIndexBuffer( new Buffer( Buffer.ELEMENT_BUFFER ).setData( [ 0, 1, 2, 0, 3, 1 ] ) );
 21 
 22     this.globalUniformCache = {
 23         Time: Date.now(),
 24         ProjectionMatrix: new Matrix4(),
 25         InverseProjectionMatrix: new Matrix4(),
 26         ViewMatrix: new Matrix4(),
 27         WorldMatrix: new Matrix4(),
 28         ViewProjectionMatrix: new Matrix4(),
 29         WorldViewMatrix: new Matrix4(),
 30         WorldViewProjectionMatrix: new Matrix4()
 31     };
 32 }
 33 
 34 
 35 RenderManager.prototype = {
 36     constructor: RenderManager,
 37     addPostProcessEffect: function( material ) {
 38         this.postProcess = true;
 39         this.postProcessEffects.push( material );
 40         return this;
 41     },
 42     applyPostProcessEffects: function() {
 43         var i, l, effect,
 44             effects = this.postProcessEffects,
 45             quad = this.quad,
 46             renderer = this.renderer,
 47             colorTexture = this.framebuffer.colorTexture;
 48 
 49         l = effects.length;
 50 
 51         var g = this.globalUniformCache;
 52         renderer.bindFramebuffer( null );
 53         renderer.clear();
 54         for ( i = 0; i < l; ++i ) {
 55             effect = effects[ i ];
 56             for ( var engineParameter in effect.engineParameters ) {
 57                 effect.setParameter( engineParameter, g[ engineParameter ] );
 58             }
 59             effect.setParameter( 'ColorTexture', colorTexture );
 60             effect.setParameter( 'FramebufferWidth', this.framebuffer.width );
 61             effect.setParameter( 'FramebufferHeight', this.framebuffer.height );
 62             renderer.useShader( effect.getShader() );
 63             renderer.render( quad );
 64         }
 65         return this;
 66     },
 67     resize: function( width, height ) {
 68         this.renderer.setSize( width, height );
 69         this.framebuffer.setDimentions( width, height );
 70         return this;
 71     },
 72     renderScene: function( scene, camera ) {
 73         if ( this.postProcess ) {
 74             this.renderer.bindFramebuffer( this.framebuffer );
 75         }
 76 
 77         this.renderer.clear();
 78         var g = this.globalUniformCache;
 79         camera.projectionMatrix.setTo( g.ProjectionMatrix );
 80         camera.getAbsoluteInverseMatrix( g.ViewMatrix );
 81         g.ViewProjectionMatrix.set( g.ProjectionMatrix ).multiply( g.ViewMatrix );
 82         g.InverseProjectionMatrix.set( g.ProjectionMatrix ).inverse();
 83 
 84         // TODO: Draw non-transparent materials first, then transparent materials
 85         //Sort drawables by material
 86 
 87         var drawableList = scene.drawableList;
 88 
 89         drawableList.sort( function( a, b ) {
 90             return a.material.uid - b.material.uid;
 91         } );
 92 
 93         var currentMaterial = -1;
 94         var l = drawableList.length;
 95         while ( l-- ) {
 96             var currentDrawable = drawableList[ l ];
 97             currentDrawable.onBeforeRender( camera );
 98             currentDrawable.getAbsoluteMatrix( g.WorldMatrix );
 99             g.WorldViewMatrix.set( g.ViewMatrix ).multiply( g.WorldMatrix );
100             g.WorldViewProjectionMatrix.set( g.ViewProjectionMatrix ).multiply( g.WorldMatrix );
101 
102             var material = currentDrawable.material;
103             for ( var engineParameter in material.engineParameters ) {
104                 material.setParameter( engineParameter, g[ engineParameter ] );
105             }
106 
107             this.renderer.useShader( material.getShader() );
108             this.renderer.render( currentDrawable.mesh );
109         }
110         
111         if ( this.postProcess ) {
112             this.applyPostProcessEffects();
113         }
114     }
115 };
