 43 			this._minPos = minPos.slice(0);

101 	 * @returns the center point expressed as 3D vector
102 	 * @type array
103 	 **/
104 	JAABox.prototype.get_centrePos=function(){
105 		var pos = this._minPos.slice(0);

110 	JAABox.prototype.getAllPoints=function(){
111 		var center,halfSide;
112 		var points;
113 		center = this.get_centrePos();
114 		halfSide = this.get_sideLengths().slice(0);
115 		Vector3DUtil.scaleBy(halfSide, 0.5);
116 		points = [];
117 		points[0] = Vector3DUtil.add(center,[halfSide[0], -halfSide[1], halfSide[2]]);
118 		points[1] = Vector3DUtil.add(center,[halfSide[0], halfSide[1], halfSide[2]]);
119 		points[2] = Vector3DUtil.add(center,[-halfSide[0], -halfSide[1], halfSide[2]]);
120 		points[3] = Vector3DUtil.add(center,[-halfSide[0], halfSide[1], halfSide[2]]);
121 		points[4] = Vector3DUtil.add(center,[-halfSide[0], -halfSide[1], -halfSide[2]]);
122 		points[5] = Vector3DUtil.add(center,[-halfSide[0], halfSide[1], -halfSide[2]]);
123 		points[6] = Vector3DUtil.add(center,[halfSide[0], -halfSide[1], -halfSide[2]]);
124 		points[7] = Vector3DUtil.add(center,[halfSide[0], halfSide[1], -halfSide[2]]);
125                         
126 		return points;
127 	}

168 	 * @param {array} pos a 3D vector

171 	JAABox.prototype.addPoint=function(pos){
172 		var _minPos=this._minPos;
173 		var _maxPos=this._maxPos;
174 		if (pos[0] < _minPos[0]) _minPos[0] = pos[0] - JNumber3D.NUM_TINY;
175 		if (pos[0] > _maxPos[0]) _maxPos[0] = pos[0] + JNumber3D.NUM_TINY;
176 		if (pos[1] < _minPos[1]) _minPos[1] = pos[1] - JNumber3D.NUM_TINY;
177 		if (pos[1] > _maxPos[1]) _maxPos[1] = pos[1] + JNumber3D.NUM_TINY;
178 		if (pos[2] < _minPos[2]) _minPos[2] = pos[2] - JNumber3D.NUM_TINY;
179 		if (pos[2] > _maxPos[2]) _maxPos[2] = pos[2] + JNumber3D.NUM_TINY;
180 	};

286 	 * @param {JAABox} box 
287 	 * @type boolean
288 	 **/
289 	JAABox.prototype.overlapTest=function(box){
290 		var _minPos=this._minPos;
291 		var _maxPos=this._maxPos;
292 		return ((_minPos[2] >= box.get_maxPos()[2]) ||
293 				(_maxPos[2] <= box.get_minPos()[2]) ||
294 				(_minPos[1] >= box.get_maxPos()[1]) ||
295 				(_maxPos[1] <= box.get_minPos()[1]) ||
296 				(_minPos[0] >= box.get_maxPos()[0]) ||
297 				(_maxPos[0] <= box.get_minPos()[0])) ? false : true;
298 	};

321 		var center=this.get_centrePos()

344 	JAABox.prototype.segmentAABoxOverlap=function(seg){
345 		var fst = 0;
346 		var fet = 1;  
347 		var pt1 = seg.origin;
348 		var pt2 = seg.getEnd();
349 		var min= this._minPos;
350 		var max = this._maxPos;
351 	  
352 		for (var i = 0; i < 3; i++) {  
353 			var s=pt1[i];var e=pt2[i];
354 			var amax=max[i];var amin=min[i];
355 			if (s < e) {  
356 				if (s > amax || e < amin){
357 					return false;  
358 				}
359 			}else {  
360 				if (e > amax || s < amin){ 
361 					return false;  
362 				}
363 			}  
364 			var d = e - s;  
365 			var st = (s < amin) ? (amin - s) / d : 0;  
366 			var et = (e > amax) ? (amax - s) / d : 1;  
367 	  
368 			if (st > fst) fst = st;  
369 			if (et < fet) fet = et;  
370 			if (fet < fst){
371 				return false;  
372 			}
373 		}  
374 
375 		return true;  
376 	}
